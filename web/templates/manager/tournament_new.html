Full format reference 
# One tournament per file. Blank lines ignored.

Name: <string>
Type: round robin | single elimination | double elimination
  (snake_case ids also work, e.g. single_elimination)
Battle Format: <showdown format>   # e.g. gen9randombattle
Best Of: Bo1 | Bo3 | 3 | …            # must be an odd positive number
Single Elim Bracket: compact | power_of_two   # optional; elim types only (default compact)

Participants:
  <provider>, <model>, <persona_slug> [, <seed>]
  # or: provider | model | persona [| seed]
  # provider: anthropic | deepseek | openrouter
  # persona_slug: filename in /personas (e.g. aggro, stall)
